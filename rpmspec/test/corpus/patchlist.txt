===============================================================================
Patchlist (Empty)
===============================================================================

%patchlist

-------------------------------------------------------------------------------

(spec
  (patchlist))

===============================================================================
Patchlist (Single patch file)
===============================================================================

%patchlist
fix-build.patch

-------------------------------------------------------------------------------

(spec
  (patchlist
    (file)))

===============================================================================
Patchlist (Multiple patch files)
===============================================================================

%patchlist
fix-build.patch
fix-tests.patch

-------------------------------------------------------------------------------

(spec
  (patchlist
    (file)
    (file)))

===============================================================================
Patchlist (URL patch)
===============================================================================

%patchlist
https://example.com/security-fix.patch

-------------------------------------------------------------------------------

(spec
  (patchlist
    (url)))

===============================================================================
Patchlist (Mixed files and URLs)
===============================================================================

%patchlist
fix-build.patch
https://example.com/security-fix.patch
local-fix.patch

-------------------------------------------------------------------------------

(spec
  (patchlist
    (file)
    (url)
    (file)))

===============================================================================
Patchlist (File with macros)
===============================================================================

%patchlist
%{name}-%{version}-fix.patch

-------------------------------------------------------------------------------

(spec
  (patchlist
    (file
      (macro_expansion
        (identifier))
      (macro_expansion
        (identifier)))))

===============================================================================
Patchlist (with %if conditional)
===============================================================================

%patchlist
fix-build.patch
%if 0%{?fedora}
fedora-specific.patch
%endif
common-fix.patch

-------------------------------------------------------------------------------

(spec
  (patchlist
    (file)
    condition: (concatenation
      (integer)
      (macro_expansion
        (conditional_expansion
          condition: (identifier))))
    consequence: (file)
    (file)))
