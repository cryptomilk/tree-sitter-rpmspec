===============================================================================
Sourcelist (Empty)
===============================================================================

%sourcelist

-------------------------------------------------------------------------------

(spec
  (sourcelist
    (section_name)))

===============================================================================
Sourcelist (Single URL)
===============================================================================

%sourcelist
https://example.com/foo-1.0.tar.gz

-------------------------------------------------------------------------------

(spec
  (sourcelist
    (section_name)
    (url)))

===============================================================================
Sourcelist (Multiple URLs)
===============================================================================

%sourcelist
https://example.com/foo-1.0.tar.gz
https://example.com/foo-data-1.0.zip

-------------------------------------------------------------------------------

(spec
  (sourcelist
    (section_name)
    (url)
    (url)))

===============================================================================
Sourcelist (URL with macros)
===============================================================================

%sourcelist
https://example.com/%{name}-%{version}.tar.gz

-------------------------------------------------------------------------------

(spec
  (sourcelist
    (section_name)
    (url
      (macro_expansion
        (identifier))
      (macro_expansion
        (identifier)))))

===============================================================================
Sourcelist (Local file path)
===============================================================================

%sourcelist
foo-1.0.tar.gz

-------------------------------------------------------------------------------

(spec
  (sourcelist
    (section_name)
    (file)))

===============================================================================
Sourcelist (Mixed URLs and files)
===============================================================================

%sourcelist
https://example.com/foo-1.0.tar.gz
local-patch.tar.gz
ftp://ftp.example.org/data.zip

-------------------------------------------------------------------------------

(spec
  (sourcelist
    (section_name)
    (url)
    (file)
    (url)))

===============================================================================
Sourcelist (with %if conditional)
===============================================================================

%sourcelist
https://example.com/foo-1.0.tar.gz
%if 0%{?fedora}
https://example.com/fedora-extra.tar.gz
%endif
https://example.com/common.tar.gz

-------------------------------------------------------------------------------

(spec
  (sourcelist
    (section_name)
    (url)
    condition: (concatenation
      (integer)
      (macro_expansion
        (conditional_expansion
          condition: (identifier))))
    consequence: (url)
    (url)))
